Detailed Description

Base class for all OpenGL based views.

You can derive from this class to render with OpenGL into your view. An Example looks like this:

@implementation MyOpenGLView : GSOpenGLView
- (void)drawRect:(CGRect)aRect
{
var gl = [[CPGraphicsContext currentContext] graphicsPort];
// Your Render Code
}
@end

You can also access VN Maker's Basic Engine:

@implementation MyOpenGLView : GSOpenGLView
{
// A gs.Quad to display a colored square in our view.
Quad _quad;
}
- (id)initWithFrame:(CGRect)aFrame
{
self = [super initWithFrame:aFrame];
// Create a new quad graphic object
_quad = new gs.Quad();
// Set a red color for it
_quad.color = new gs.Color(255, 0, 0);
// Make it 100x100 pixel in size and display it at 10,10 in our view.
_quad.rect.set(10, 10, 100, 100);
return self;
}
- (void)drawRect:(CGRect)aRect
{
// Assign the renderer-object for this open-gl view as the basic engine's renderer.
gs.Graphics.renderer = _renderer;
// Draw the quad. This will add a render-task but not immediately render the quad.
_quad.draw();
// Render all render-tasks. This will render the quad.
_renderer.render();
}
@end
